        function toggleTheme() {
            const currentTheme = document.body.classList.contains('theme-dark') ? 'dark' : 'light';
            const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
            setTheme(newTheme);
        }
        
        function setTheme(theme) {
            document.body.classList.remove('theme-light', 'theme-dark');
            document.body.classList.add(`theme-${theme}`);
            
            const themeToggle = document.getElementById('themeToggle');
            if (theme === 'dark') {
                themeToggle.innerHTML = '<i class="fas fa-sun"></i>';
                themeToggle.title = 'Переключить на светлую тему';
            } else {
                themeToggle.innerHTML = '<i class="fas fa-moon"></i>';
                themeToggle.title = 'Переключить на темную тему';
            }
            
            localStorage.setItem('theme', theme);
        }
        
        function toggleFaq(element) {
            const answer = element.nextElementSibling;
            const icon = element.querySelector('i');
            
            if (answer.style.display === 'block') {
                answer.style.display = 'none';
                icon.className = 'fas fa-chevron-down';
            } else {
                answer.style.display = 'block';
                icon.className = 'fas fa-chevron-up';
            }
        }

        function detectCardType(cardNumber) {
            cardNumber = cardNumber.replace(/\s/g, '');
            
            if (cardNumber.length < 1) return;
            
            let cardType = 'mir'; 
            
            if (cardNumber.startsWith('4')) {
                cardType = 'visa';
            } else if (cardNumber.startsWith('5')) {
                cardType = 'mastercard';
            } else if (cardNumber.startsWith('2') || cardNumber.startsWith('6')) {
                cardType = 'mir';
            }
            
            selectCardType(cardType);
        }

        function processPayment() {
            const contractNumber = document.getElementById('contractNumber').value;
            const amount = document.getElementById('amount').value;
            const email = document.getElementById('payerEmail').value;
            const agreeTerms = document.getElementById('agreeTerms').checked;
            
            const successMsg = document.getElementById('paymentSuccess');
            const errorMsg = document.getElementById('paymentError');
            
            successMsg.style.display = 'none';
            errorMsg.style.display = 'none';
            
            
            if (!contractNumber) {
                showPaymentError('Пожалуйста, введите номер договора или лицевого счета');
                return;
            }
            
            if (!amount || amount <= 0) {
                showPaymentError('Пожалуйста, введите корректную сумму для оплаты');
                return;
            }
            
            if (!agreeTerms) {
                showPaymentError('Необходимо согласие с условиями обработки персональных данных');
                return;
            }
            
            
            let paymentDataValid = true;
            let paymentDetails = {};
            
            if (selectedPaymentMethod === 'card') {
                const cardNumber = document.getElementById('cardNumber').value.replace(/\s/g, '');
                const cardHolder = document.getElementById('cardHolder').value;
                const cardExpiry = document.getElementById('cardExpiry').value;
                const cardCVC = document.getElementById('cardCVC').value;
                
                if (!cardNumber || cardNumber.length < 16) {
                    showPaymentError('Некорректный номер карты');
                    return;
                }
                
                if (!cardHolder) {
                    showPaymentError('Введите имя держателя карты');
                    return;
                }
                
                if (!cardExpiry || cardExpiry.length < 5) {
                    showPaymentError('Введите срок действия карты (ММ/ГГ)');
                    return;
                }
                
                if (!cardCVC || cardCVC.length < 3) {
                    showPaymentError('Введите CVC/CVV код');
                    return;
                }
                
                paymentDetails = {
                    method: 'card',
                    cardType: getCardTypeName(selectedCardType),
                    cardNumber: cardNumber.substring(0, 4) + ' **** **** ' + cardNumber.substring(12),
                    cardHolder: cardHolder,
                    cardExpiry: cardExpiry
                };
                
            } else if (selectedPaymentMethod === 'wallet') {
                let walletNumber = '';
                
                if (selectedWalletOption === 'yoomoney') {
                    walletNumber = document.getElementById('yoomoneyNumber').value;
                    if (!walletNumber) {
                        showPaymentError('Введите номер кошелька ЮMoney');
                        return;
                    }
                } else if (selectedWalletOption === 'webmoney') {
                    walletNumber = document.getElementById('webmoneyNumber').value;
                    if (!walletNumber) {
                        showPaymentError('Введите номер кошелька WebMoney');
                        return;
                    }
                } else if (selectedWalletOption === 'qiwi') {
                    walletNumber = document.getElementById('qiwiNumber').value;
                    if (!walletNumber) {
                        showPaymentError('Введите номер телефона QIWI');
                        return;
                    }
                }
                
                paymentDetails = {
                    method: 'wallet',
                    walletType: getWalletTypeName(selectedWalletOption),
                    walletNumber: walletNumber
                };
                
            } else if (selectedPaymentMethod === 'mobile') {
                const mobilePhone = document.getElementById('mobilePhone').value;
                
                if (!mobilePhone) {
                    showPaymentError('Введите номер телефона');
                    return;
                }
                
                paymentDetails = {
                    method: 'mobile',
                    operator: getOperatorName(selectedMobileOperator),
                    phone: mobilePhone
                };
            }
            
            console.log('Отправка оплаты:');
            console.log('Номер договора:', contractNumber);
            console.log('Сумма:', amount);
            console.log('Email:', email);
            console.log('Способ оплаты:', getPaymentMethodName(selectedPaymentMethod));
            console.log('Детали оплаты:', paymentDetails);
            
           
            const transactionId = 'TXN-' + Math.random().toString(36).substr(2, 9).toUpperCase();
            
           
            const successContent = document.getElementById('paymentSuccessContent');
            successContent.innerHTML = `
                <strong>Оплата успешно отправлена!</strong><br>
                Номер договора: ${contractNumber}<br>
                Сумма: ${amount} руб.<br>
                Способ оплаты: ${getPaymentMethodName(selectedPaymentMethod)}<br>
                ${paymentDetails.cardType ? 'Тип карты: ' + paymentDetails.cardType + '<br>' : ''}
                ${paymentDetails.walletType ? 'Тип кошелька: ' + paymentDetails.walletType + '<br>' : ''}
                ${paymentDetails.operator ? 'Оператор: ' + paymentDetails.operator + '<br>' : ''}
                ${email ? 'Чек будет отправлен на: ' + email + '<br>' : ''}
                Дата: ${new Date().toLocaleString('ru-RU')}<br>
                <small>Номер транзакции: ${transactionId}</small>
            `;
            
            successMsg.style.display = 'flex';
            successMsg.scrollIntoView({ behavior: 'smooth', block: 'center' });

                document.getElementById('contractNumber').value = '';
                document.getElementById('amount').value = '';
                document.getElementById('payerEmail').value = '';
                
                
                if (selectedPaymentMethod === 'card') {
                    document.getElementById('cardNumber').value = '';
                    document.getElementById('cardHolder').value = '';
                    document.getElementById('cardExpiry').value = '';
                    document.getElementById('cardCVC').value = '';
                }
                
               
                if (selectedPaymentMethod === 'wallet') {
                    document.getElementById('yoomoneyNumber').value = '';
                    document.getElementById('webmoneyNumber').value = '';
                    document.getElementById('qiwiNumber').value = '';
                }

                    document.getElementById('mobilePhone').value = '';
                }

        }

        function initCarousel() {
            const slides = document.querySelectorAll('.carousel-slide');
            if (slides.length === 0) return;
            
            totalSlides = slides.length;
            

            const prevBtn = document.querySelector('.prev-btn');
            const nextBtn = document.querySelector('.next-btn');
            const indicators = document.querySelectorAll('.carousel-indicator');
            const playBtn = document.getElementById('playCarousel');
            const pauseBtn = document.getElementById('pauseCarousel');
            
            if (prevBtn) {
                prevBtn.addEventListener('click', showPrevSlide);
            }
            
            if (nextBtn) {
                nextBtn.addEventListener('click', showNextSlide);
            }
            
            if (indicators) {
                indicators.forEach((indicator, index) => {
                    indicator.addEventListener('click', () => {
                        goToSlide(index);
                    });
                });
            }
            
            if (playBtn) {
                playBtn.addEventListener('click', startAutoSlide);
            }
            
            if (pauseBtn) {
                pauseBtn.addEventListener('click', pauseAutoSlide);
            }

            startAutoSlide();
            

            const carousel = document.querySelector('.geography-carousel');
            if (carousel) {
                carousel.addEventListener('mouseenter', pauseAutoSlide);
                carousel.addEventListener('mouseleave', () => {
                    if (isAutoSlide) {
                        startAutoSlide();
                    }
                });
            }
        }
        
        function showNextSlide() {
            currentSlide = (currentSlide + 1) % totalSlides;
            updateCarousel();
        }
        
        function showPrevSlide() {
            currentSlide = (currentSlide - 1 + totalSlides) % totalSlides;
            updateCarousel();
        }
        
        function goToSlide(slideIndex) {
            currentSlide = slideIndex;
            updateCarousel();
        }
        
        function updateCarousel() {
            const slides = document.querySelectorAll('.carousel-slide');
            slides.forEach((slide, index) => {
                slide.classList.remove('active', 'previous');
                
                if (index === currentSlide) {
                    slide.classList.add('active');
                } else if (index === (currentSlide - 1 + totalSlides) % totalSlides) {
                    slide.classList.add('previous');
                }
            });
            

            const indicators = document.querySelectorAll('.carousel-indicator');
            indicators.forEach((indicator, index) => {
                indicator.classList.remove('active');
                if (index === currentSlide) {
                    indicator.classList.add('active');
                }
            });
        }
        
        function startAutoSlide() {

            if (autoSlideInterval) {
                clearInterval(autoSlideInterval);
            }
            
  
            autoSlideInterval = setInterval(showNextSlide, slideIntervalTime);
            isAutoSlide = true;
            
 
            const playBtn = document.getElementById('playCarousel');
            const pauseBtn = document.getElementById('pauseCarousel');
            
            if (playBtn) playBtn.style.display = 'none';
            if (pauseBtn) pauseBtn.style.display = 'flex';
        }
        
        function pauseAutoSlide() {
            if (autoSlideInterval) {
                clearInterval(autoSlideInterval);
                autoSlideInterval = null;
            }
            isAutoSlide = false;
            

            const playBtn = document.getElementById('playCarousel');
            const pauseBtn = document.getElementById('pauseCarousel');
            
            if (playBtn) playBtn.style.display = 'flex';
            if (pauseBtn) pauseBtn.style.display = 'none';
        }
